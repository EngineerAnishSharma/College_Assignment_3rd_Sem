#include <stdio.h>
#include <stdlib.h>
struct node{
    int data;
    struct node* next;
};
struct node*temp,*newnode;
struct node*head=NULL;
void insertAtEnd(){
    int x;
    newnode=(struct node*)malloc(sizeof(struct node));
    printf("\nEnter a data of created node:\n");
    scanf("%d",&x);
    newnode->data=x;
    newnode->next=NULL;
    if(head==NULL){
        head=temp=newnode;
    }
    else{
        temp->next=newnode;
        temp=newnode;
    }    
}
void insertAtStart(){
    int x;
    newnode=(struct node*)malloc(sizeof(struct node));
    printf("\nEnter a data of created node:\nqaws");
    scanf("%d",&x);
    newnode->data=x;
    newnode->next=head;
    head=newnode;
}

void delAtEnd(){
    temp=head;
    struct node*t=head;
    while(temp->next!=NULL){
        t=temp;
        temp=temp->next;
    }
    printf("\nDeleted data:%d\n",temp->data);
    t->next=0;

}
void delAtStart(){
    printf("Delete first node:%d",head->data);
    head=head->next;
}
void display(){
    temp=head;
    while(temp->next!=0){
        printf("%d ",temp->data);
        temp=temp->next;
    }
    printf("%d\n",temp->data);
}
int main()
{
    int choice;
    while(1){
        printf("\nInsertion at End- 1\nInsertion at starting - 2\nDisplaying data - 3\nDelete at End - 4\nDelete at Start - 5\nExit - 0\n");
        scanf("%d",&choice);
        if(choice==1){
            insertAtEnd();
        }
        else if(choice==2){
            insertAtStart();
        }
        else if(choice==3){
            display();
        }
        else if(choice==4){
            delAtEnd();
        }
        else if(choice==5){
            delAtStart();
        }
        else if(choice==0){
            break;
        }
        
    }
    return 0;
}
